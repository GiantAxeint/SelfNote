#### 目的
虚函数的目的，如[[9.继承]]中所说的问题一样，我们来看以下代码：
```
class Entity 
{
public:
	virtual std::string GetName() { return "Entity"; }
	

};
class Play : public Entity
{
private:
	std::string m_name;
public:
	Play(std::string name) 
		:m_name(name) {};
	std::string GetName() override { return m_name; }
};
```
int main()
{
	Entity* e = new Entity();
	Play*   p = new Play();
}
